
Database :`billing`

CREATE TABLE product (
pid NUMBER(4) PRIMARY KEY,
pname VARCHAR2(30) ,
price NUMBER(5,2) ,
stock NUMBER(4) 
);

CREATE TABLE bill(
  billno NUMBER(4) PRIMARY KEY ,
  customer VARCHAR2(30) ,
  date DATETIME ,
  total NUMBER(5,2)
  );
  
CREATE TABLE billproducts (
  billno NUMBER(4) REFERENCES (bill.billno) ,
  pid NUMBER(4) REFERENCES (product.pid)
  );
